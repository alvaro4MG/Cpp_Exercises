int funcWithStatic()
{
    static int i = 10;
    i *= 10;
    return i;
}

int main()
{
    int value;

    value = funcWithStatic();
    value = funcWithStatic();

    cout << value;

    return 0;
}

int main()
{
    int a = 1;
    int b = 1;

    int* pa = &a;
    int* pb = &b;

    if(pa == pb) {
	cout << "true";
    } else {
	cout << "false";
    }

    return 0;
}
