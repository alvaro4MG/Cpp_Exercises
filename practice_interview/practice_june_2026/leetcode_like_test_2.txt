    #include <unordered_set>
    #include <vector>
    bool checkRepeated(const std::vector<int>& nums) {
        std::unordered_set<int> set;

        for(const auto& v : nums){
            if(set.contains(v)){
                return true;
            }else{
                set.insert(v);
            }
        }
        return false;
    }

    #include <string>
    void newReverse(string& message){
        int i = 0;
        int j = message.size() - 1;
        char aux;	// or use std::swap(message[i], message[j]);

        while(i < j){
            aux = message[i];
            message[i] = message[j];
            message[j] = aux;
            i++;
            j--;
        }
    }

    void twoSum(const vector<int>& nums, const int& target, int& id1, int& id2){
        id1 = 0;
        id2 = nums.size() - 1;
        int sum = nums[id1] + nums[id2];

        while(sum != target){
            if(sum < target){
                id1++;
            }else{
                id2--;
            }

            sum = nums[id1] + nums[id2];
        }
    }

    #include <stack>
    #include <string>
    bool validParenthesis(const string& message){
        std::stack<char> pila;

        for(auto& c : message){
            if(c == '(' || c== '[' || c == '{'){
                pila.push(c);
            }else{
		if(pila.empty()){
		    return false;
		}

                switch(c){
                    case ')':
                        if(pila.top() != '('){
                            return false;
                        }
                        break;
                    case ']':
                        if(pila.top() != '['){
                            return false;
                        }
                        break;
                    case '}':
                        if(pila.top() != '{'){
                            return false;
                        }
                        break;
                }
                pila.pop();
            }
        }
        return pila.empty();
    }
